function insertion(arr) {
  for (let i = 0; i < arr.length; i++) {
    let j = i

    while (j > 0 && arr[j] < arr[j - 1]) {
      let temp = arr[j - 1]

      arr[j - 1] = arr[j]
      arr[j] = temp

      j = j - 1
    }
  }

  return arr
}
